    <task_id>|<title>|<status>|<creator>|<assignee>|<priority>|<created_timestamp>|<ttl_in_seconds>

    possible statuses:
      - 1=created

      - 3=inprogress
      - 4=completed
      - 5=expired

    possible priorities:
      - 1=low
      - 2=medium
      - 3=high
